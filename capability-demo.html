// Example: Using the new capabilities service
import { 
    getAllCapabilities, 
    getCapabilityPresets, 
    validateCapabilities,
    suggestCapabilities 
} from './services/capabilities-service.js';

// Get all available capabilities
const capabilities = getAllCapabilities();

// Apply a preset
const preset = getCapabilityPresets().find(p => p.name === 'Advanced AI');
model.capabilities = preset.capabilities;

// Validate capabilities
const validation = validateCapabilities(model.capabilities);
if (!validation.isValid) {
    console.log('Errors:', validation.errors);
    console.log('Warnings:', validation.warnings);
}

// Auto-suggest capabilities
const suggestions = suggestCapabilities('GPT-4 Vision', 'Multimodal AI model', 'openai');
// Returns: ['text', 'code', 'vision', 'reasoning']
